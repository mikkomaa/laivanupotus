3   import java.awt.Color;
4   import java.awt.event.ActionEvent;
5   import java.awt.event.ActionListener;
6   import laivanupotus.domain.Ruutu;
7   import static laivanupotus.domain.Ruutu.*;
8   import laivanupotus.sovelluslogiikka.Peli;
9   
10  /**
11   * Luokka m채채rittelee pelaajan graafisen pelilaudan ruutujen toiminnallisuuden,
12   * eli pelaaja voi asettaa laudalleen laivat hiirell채 ennen peli채
13   */
14  public class PelaajanLaudanKuuntelija implements ActionListener {
15  
16      private final Peli peli;
17      private final Painike painike;
18      private final int korkeus;
19      private final int leveys;
20  
21      /**
22       * Luokan konstruktori
23       * 
24       * @param painike Pelilaudan ruutua vastaava painike
25       * @param peli Sovelluslogiikan peliolio
26       */
27      public PelaajanLaudanKuuntelija(Painike painike, Peli peli) {
28          this.peli = peli;
29          this.painike = painike;
30          this.korkeus = painike.getKorkeus();
31          this.leveys = painike.getLeveys();
32      }
33  
34      @Override
35      public void actionPerformed(ActionEvent ae) {
36          if (this.peli.onkoPeliKaynnissa()) {
37              return;
38          }
39  
40          Ruutu ruutu = this.peli.getPelaajanRuutu(korkeus, leveys);
41          if (ruutu == TYHJA) {
42              this.peli.setPelaajanRuutu(korkeus, leveys, LAIVA);
43              this.painike.setBackground(Color.lightGray);
44          } else { // ruutu == LAIVA
45              this.peli.setPelaajanRuutu(korkeus, leveys, TYHJA);
46              this.painike.setBackground(Color.white);
47          }
48      }
49  }
