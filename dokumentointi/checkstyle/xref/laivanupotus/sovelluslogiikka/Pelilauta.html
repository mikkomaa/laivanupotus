16      /**
17       * Luokan konstruktori, joka luo pelilaudan ja alustaa sen peli채 varten
18       */
19      public Pelilauta() {
20          this.lauta = new Ruutu[LAUDAN_KORKEUS][LAUDAN_LEVEYS];
21          this.pisteidenLaskija = new PisteidenLaskija();
22          this.alustaLauta();
23      }

25      /**
26       * Metodi alustaa pelilaudan uutta peli채 varten
27       */
28      public void alustaLauta() {
29          for (int korkeus = 0; korkeus < LAUDAN_KORKEUS; korkeus++) {
30              for (int leveys = 0; leveys < LAUDAN_LEVEYS; leveys++) {
31                  this.lauta[korkeus][leveys] = TYHJA;
32              }
33          }
34          this.pisteidenLaskija.nollaaPisteet();
35      }
36  
37      /**
38       * Metodi asettaa pelilaudan tietyn ruudun arvon
39       * 
40       * @param korkeus Asetettavan ruudun y-koordinaatti
41       * @param leveys Asetettavan ruudun x-koordinaatti
42       * @param ruutu Ruudulle asetettava arvo
43       */
44      public void setRuutu(int korkeus, int leveys, Ruutu ruutu) {
45          if (onkoRuutuLaudalla(korkeus, leveys)) {
46              this.lauta[korkeus][leveys] = ruutu;
47          }
48      }
49  
50      /**
51       * Metodi palauttaa pelilaudan tietyn ruudun arvon
52       * 
53       * @param korkeus Palautettavan ruudun y-koordinaatti
54       * @param leveys Palautettavan ruudun x-koordinaatti
55       * 
56       * @return Palauttaa ruudun arvon
57       */
58      public Ruutu getRuutu(int korkeus, int leveys) {
59          if (onkoRuutuLaudalla(korkeus, leveys)) {
60              return this.lauta[korkeus][leveys];
61          }
62          return null;
63      }
64      
65      /**
66       * Metodi tarkistaa, ovatko laivat oikein pelilaudalla
67       * 
68       * @return Palauttaa true, jos laivat ovat oikein, muuten false
69       */
70      public boolean lueLaivat() {
71          LaivojenLukija lukija = new LaivojenLukija();
72          return lukija.lueLaivat(this);
73      }
74  
75      /**
76       * Metodi muuttaa pelilaudan ruudun arvoksi AMMUTTU, jos ruudussa ei ole
77       * laivaa, ja arvoksi OSUMA, jos ruudussa on laiva
78       * 
79       * @param korkeus Muutettavan ruudun y-koordinaatti
80       * @param leveys Muutettavan ruudun x-koordinaatti
81       * 
82       * @return Palauttaa muutetun ruudun arvon muutoksen j채lkeen
83       */
84      public Ruutu ammu(int korkeus, int leveys) {
85          Ruutu ruutu = getRuutu(korkeus, leveys);
86          if (ruutu == TYHJA) {
87              setRuutu(korkeus, leveys, AMMUTTU);
88          } else if (ruutu == LAIVA) {
89              this.setRuutu(korkeus, leveys, OSUMA);
90          }
91          return getRuutu(korkeus, leveys);
92      }
93  
94      /**
95       * Metodi tarkistaa, ovatko pelilaudan laivat uponneet eli jokainen
96       * laivaruutu saanut osuman
97       * 
98       * @return Palauttaa true, jos laivat ovat uponneet, muuten false
99       */
100     public boolean ovatkoLaivatUponneet() {
101         int osumia = 0;
102         for (int korkeus = 0; korkeus < LAUDAN_KORKEUS; korkeus++) {
103             for (int leveys = 0; leveys < LAUDAN_LEVEYS; leveys++) {
104                 if (this.lauta[korkeus][leveys] == OSUMA) {
105                     osumia++;
106                 }
107             }
108         }
109         return osumia == LAIVARUUTUJA;
110     }

116      */
117     public void lisaaPisteet(int maara) {
118         this.pisteidenLaskija.lisaaPisteet(maara);
119     }
120     
121     public int getPisteet() {
122         return this.pisteidenLaskija.getPisteet();
123     }
124     
125     /**
126      * Metodi nollaa pelilaudan pisteet

129         this.pisteidenLaskija.nollaaPisteet();
130     }
131     
132     @Override
133     public String toString() {
134         String mjono = "";
135         for (int korkeus = 0; korkeus < LAUDAN_KORKEUS; korkeus++) {
136             for (int leveys = 0; leveys < LAUDAN_LEVEYS; leveys++) {
137                 mjono += this.lauta[korkeus][leveys] + "\t";
138             }
139             mjono += "\n";
140         }
141         return mjono;
142     }
143     
144     private boolean onkoRuutuLaudalla(int korkeus, int leveys) {
145         return korkeus >= 0 && korkeus < LAUDAN_KORKEUS && leveys >= 0 && leveys < LAUDAN_LEVEYS;
146     }
